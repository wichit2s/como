รูปแบบที่ ๒¶

โจทย์ให้รับข้อมูลตามจำนวนที่ระบุไว้ในแต่ละกรณี เช่น

จงเขียนโปรแกรมเพื่อแสดงตัวเลขในลำดับของจำนวนเต็มที่ผู้ใช้กรอกทั้งหมด โดยบรรทัดแรกผู้ใช้จะระบุจำนวนเต็ม n ที่จะกรอก และ n บรรทัดถัดมาเป็นจำนวนเต็มในลำดับหนึ่งบรรทัดต่อหนึ่งจำนวน และบรรทัดสุดท้ายเป็นจำนวนเต็มระบุตำแหน่งในลำดับที่ต้องการให้แสดง

#include <bits/stdc++.h>

using namespace std;

int main() {

  int x[6] = { 0, 1, 2, 3, 4, 5 };
  double a[6];
  a[0] = 3.45;
  a[1] = 3.65;
  a[2] = a[0] + 0.15;
  a[x[3]] = a[0]*x[2];

  a[4] = a[3] - a[2];
  a[5] = a[4] - a[3];

  cout << a[2] << endl;
  cout << a[1] << ", " << a[2] << ", " << a[3] << endl;
  return 0;
}

#include <bits/stdc++.h>

using namespace std;

float sum(vector<int> a) {
  float sum = 0;
  for (auto i=0; i<a.size(); i++) {
    sum += a[i];
  }
  return sum;
}

float sum2(vector<int> a) {
  float sum = 0;
  for (auto x : a) {
    sum += x;
  }
  return sum;
}

float sum3(vector<int> a) {
  float sum = 0;
  for (auto i=a.begin(); i != a.end(); i++) {
    sum += *i;
  }
  return sum;
}

float sum4(vector<int> a) {
  float sum = 0;
  auto i = a.begin();
  while (i != a.end()) {
    sum += *i;
    i++;
  }
  return sum;
}

void print(vector<string> names) {
  for (auto name : names) {
    cout << setw(20) << name ;
  }
  cout << endl;
}

int main() {

   vector<int> b { 2, 4, 7, 9, 11 };
   cout << sum(b) << endl;
   
   for (auto i=1; i<=10; i+=2) {
     b.push_back(i);
   }
   cout << sum2(b) << endl;
   cout << sum3(b) << endl;
   cout << sum4(b) << endl;

   vector<string> names { u8"มานะ", u8"มานี", u8"ปิติ", u8"ชูใจ" };
   print(names);

  return 0;
}
