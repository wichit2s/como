#include <bits/stdc++.h>

using namespace std;

int main() {

  for (auto i=0; i < 10; i++) {
    cout << setw(5) << i << setw(10) << i*i << setw(10) << i*i*i << endl;
  }

  cout << endl;
  auto i = 0;
  for (; i < 10; i++) {
    cout << setw(5) << i << setw(10) << i*i << setw(10) << i*i*i << endl;
  }
  
  cout << endl;
  auto j = 0;
  for (; j < 10; ){
    cout << setw(5) << j << setw(10) << j*j << setw(10) << j*j*j << endl;
    j++;
  }

  auto x = {10, 20, 30, 40, 50, 60};
  for (auto e : x) {
    cout << e << setw(10) << e*e*e*e << endl;
  }

  auto names = {"Anthony", "Betty", "Cathy", "Dorothy", "Eddy", "Franky"};
  for (auto name : names) {
    cout << name << endl;
  }

  return 0;
}
