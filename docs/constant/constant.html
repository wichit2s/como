const float E = 2.71828182845f;
const long double PI = 3.14159265358979L;
const long long MAX = 10000000l;
const int MIN = -10000000;

int main() {

  cout << E << endl;
  cout << PI << endl;
  cout << MAX << endl;
  cout << MIN << endl;

  cout.precision(18);
  cout << E << endl;
  cout << PI << endl;

  return 0;
}

#include <bits/stdc++.h>

using namespace std;

#define E   2.71828182845f
#define PI  3.14159265358979L
#define MAX 10000000l
#define MIN -10000000

#define NL  '\n'

int main() {

  cout << E << NL;
  cout << PI << NL;
  cout << MAX << NL;
  cout << MIN << NL;

  cout.precision(18);
  cout << E << NL;
  cout << PI << NL;

  return 0;
}
