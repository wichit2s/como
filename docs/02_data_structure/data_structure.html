#include <bits/stdc++.h>

using namespace std;

int main() {
  tuple<string, double, int, int, int> andy("Andy Archer", 3.22, 10, 21, 23);
  auto barby = make_tuple("Barby Blossom", 3.44, 12, 18, 24);

  cout << get<0>(andy) << " : " << get<1>(andy) << endl;

  get<1>(andy) = 3.33;
  cout << get<0>(andy) << " : " << get<1>(andy) << endl;

  string andy_name;
  float andy_gpa;
  tie(andy_name, andy_gpa, ignore, ignore, ignore) = andy;
  cout << andy_name << " : " << andy_gpa << endl;

  int x, y, z;
  tie(ignore, ignore, x, y, z) = barby;

  get<1>(barby) = get<1>(andy);
  cout << get<0>(barby) << " : " << " " << (x + y + z) / 2.0 << endl;

  cout << "andy size >> " << tuple_size<decltype(andy)>::value << endl;
  return 0;
}

#include <bits/stdc++.h>

using namespace std;

float sum(array<int, 5> a) {
  float sum = 0;
  for (auto x : a) {
    sum += x;
  }
  return sum;
}

void print(array<int, 5> a) {
  for (auto x : a) {
    cout << x << " ";
  }
  cout << endl;
}

int revert(int a, int b) {
  return (a > b)? -1 : (a == b)? 0 : 1;
}

int main() {

   array<int, 5> b { 12, 8, 7, 9, 11 };

   print(b);
   cout << "sum(b) = " << sum(b) << endl;

   b[2] = 10;
   b[0] = 13;
   print(b);

   sort(b.begin(), b.end());
   print(b);

   sort(b.begin(), b.end(), revert);
   print(b);

   cout << "b.size(): " << b.size() << endl;

  return 0;
}

#include <bits/stdc++.h>

using namespace std;

int sum(map<string, array<int,2>> a) {
  int sum = 0;
  auto it = a.begin();
  while (it != a.end()) {
    sum += it->second[0] + it->second[1];
    it++;
  }
  return sum;
}

void print(map<string, array<int,2>> a) {
  cout << "{" << endl;
  for (auto x : a) {
    cout << x.first << " => { " << x.second[0] << ", " << x.second[1] << " }," << endl;
  }
  cout << "}" << endl;
}

int main() {

   map<string, array<int, 2>> b;
   b["andy"]  = { 2, 7};
   b["barby"] = { 7, 11};
   b["cathy"] = { 5, 11};
   b["dundy"] = { 7, 10};
   b["eddy"]  = { 6, 3};
   
   print(b);
   cout << "sum: " << sum(b) << endl;
   b.erase("barby");
   print(b);
   cout << "sum: " << sum(b) << endl;

  return 0;
}
