#include <bits/stdc++.h>

using namespace std;

int main() {

  auto x = 1234;

  cout << "Hello world." << endl;
  cout << u8"สวัสดีชาวโลก\n";
  cout << "x = " << x << '\n';

  auto t = true, f = false;
  cout << "t = " << t << "  f = " << f << endl;
  cout << "t = " << t << "  f = " << boolalpha << f << endl;
  cout << "t = " << boolalpha << t << "  f = " << f << endl;

  auto y = 0.1234567890;
  auto name = "Paul";
  cout << "name = [" << name << "]  y = [" << y << "]\n";
  cout << "name = [" << setw(15) << name << "]  y = [" << setw(15) << y << "]\n";
  cout << "name = [" << setw(15) << right << name << "]  y = [" << setw(15) << left << y << "]\n";
  cout << "name = [" << setw(15) << right << name << "]  y = [" << setw(15) << left << setprecision(4) << y << "]\n";
  cout << "name = [" << setw(15) << right << setfill('*') << name << "]  y = [" << setw(15) << left << setprecision(4) << showpos << y << "]\n";
  cout << "name = [" << setw(15) << right << setfill('*') << name << "]  y = [" << scientific << y << "]\n";
  cout << "name = [" << setw(15) << right << setfill('*') << name << "]  y = [" << fixed << y << "]\n";
  cout << "name = [" << setw(15) << right << setfill('*') << name << "]  12 = [" << noshowpoint << 12.0 << "]\n";

  x = 1234;
  cout << "x = " << x << endl;
  cout << "x = " << noshowpos << x << endl;
  cout << "dec(x) = " << dec << x << endl;
  cout << "oct(x) = " << oct << x << endl;
  cout << "hex(x) = " << hex << x << endl;
  cout << "showbase oct(x) = " << showbase << oct << x << endl;
  cout << "showbase hex(x) = " << showbase << hex << x << endl;
  cout << "showbase uppercase oct(x) = " << showbase << uppercase << oct << x << endl;
  cout << "showbase hex(x) = " << showbase << hex << x << endl;

  return 0;
}

int main() {

  string line;

  string name;
  int id;
  float gpa;

  cout << u8"กรอกชื่อ: ";
  getline(cin, name);

  cout << u8"กรอกรหัส: ";
  getline(cin, line);
  stringstream(line) >> id;
  cout.flush();

  cout << u8"กรอกเกรดเฉลี่ย: ";

  return 0;
}

การแสดงผลด้วย printf¶

คำสั่งประกาศของ printf

int printf(const char *format, ... )

โดยที่ format เป็นข้อความที่สามารถนำค่าของตัวแปรมาแสดงผลได้ในต่ำแหน่งของ tag พิเศษ

รูปแบบของ tag พิเศษเป็นดังนี้

%[flags][width][.precision][length]specifier

โดยค่าที่ระบุไว้ใน [ ] ไม่จำเป็นต้องมีก็ได้ และแต่ละตำแหน่งมีความหมายและค่าที่เป็นไปได้ดังนี้

Possible Values

Meaning

flags

-

left justify

+

show + or - sign

(space)

blank space before the value.

#

o, x, X shows 0, 0x, 0X

e, E, f shows decimal points.

g, G shows decimal with trailing zeros.

width

(number)

Number of characters for display area.

*

Use preceding variable's value as width.

precision

(number)

d, i, o, u, x, X - minimum number of digits with padded zeros

. or .0 means no character printed if the value is zero.

e, E and f - number of digits after dot.

g, G - number of significant digits.

*

Use preceding variable's value as width.

length

#include <bits/stdc++.h>

int main() {

  char c = 'a';
  printf(u8"x มีค่า %c\n", c);

  int d = 20;
  printf(u8"d มีค่า|%d|%o|%#x|%X|%+d|%+10.5d|\n", d, d, d, d, d, d);

  double x = 1234567890;
  printf("%15.5e %15.5f %15.5g %15.5G\n", x, x, x, x);

  float f = 3.99;
  float *ptr = &f;
  printf("f=%f  ptr=%p  *ptr=%f \n", f, ptr, *ptr);

  printf("%n-%%\n", &d);
  return 0;
}

#include <bits/stdc++.h>

int main() {

  char c = 'a';
  scanf("%c%*c", &c);
  printf(">> %c\n", c);

  int d, o, x;
  scanf("%d%o%x", &d, &o, &x);
  printf(">> %15.5d|%15.5o|%15.5x\n", d, o, x);

  unsigned int u;
  scanf("%u", &u);
  printf("%u\n", u);

  return 0;
}
