#include <bits/stdc++.h>

using namespace std;

int main() {
  int x = 3, y = 4;

  y += x;     //  y = y + x;
  y -= x;
  y *= x;
  y /= x;
  y %= x;

  y >>= x;    //  y = y >> x;
  y <<= x;
  y &= x;
  y |= x;

  return 0;
}

#include <bits/stdc++.h>

using namespace std;

int main() {
  int x = 3, y = 4;

  x++;        // x = x+1;
  x--;        // x = x-1;

  cout << u8"ค่า x   :" << x << endl;
  cout << u8"ค่า x++ :" << x++ << endl;
  cout << u8"ค่า x-- :" << x-- << endl;
  cout << u8"ค่า x   :" << x << endl;
  cout << endl;
  cout << u8"ค่า y   :" << y << endl;
  cout << u8"ค่า ++y :" << ++y << endl;
  cout << u8"ค่า --y :" << ++y << endl;
  cout << u8"ค่า y   :" << y << endl;

  return 0;
}

using namespace std;

int main() {
  int x = 3, y = 4;
  bool z, b = true;

  z = (x == y);

  cout << "b = " << b << endl;
  cout << "z = " << z << endl;
  cout << (x == y) << endl;
  cout << (x != y) << endl;
  cout << (x > y) << endl;
  cout << (x < y) << endl;
  cout << (x >= y) << endl;
  cout << (x <= y) << endl;

  return 0;
}

#include <bits/stdc++.h>

using namespace std;

int main() {
  int x = 3, y = 4, z = 6;
  bool a, b = false;

  a = (x <= y);

  cout << "!true = " << !true << endl;
  cout << "!false = " << !false << endl;
  cout << "!a = " << !a << endl;
  cout << !(x == y) << endl;
  cout << ( (x <= y) && ( y <= z) ) << endl;
  cout << ( (x <= y) || ( y <= z) ) << endl;

  return 0;
}
