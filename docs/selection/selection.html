#include <bits/stdc++.h>

using namespace std;

int main() {
  char grade;

  cout << u8"กรอกเกรด: ";
  cin >> grade;

  switch (grade) {
    case 'A':
    case 'a':
      cout << "ได้เกรด เอ" << endl;
      break;
    case 'b':
    case 'B':
      cout << "ได้เกรด บี" << endl;
      break;
    case 'C': case 'c':
      cout << "ได้เกรด ซี" << endl;
      break;
    case 'D': case 'd':
      cout << "ได้เกรด ดี" << endl;
      break;
    default:
      cout << "ไม่ผ่าน" << endl;
      break;
  }

  return 0;
}
